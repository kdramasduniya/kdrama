        <%-
            // Split shows into new (last 30 days) and old
        -%>
        <% 
            const now = new Date();
            const THIRTY_DAYS = 1000 * 60 * 60 * 24 * 30;
            const newShows = [];
            const oldShows = [];
            if (shows && shows.length) {
                shows.forEach(s => {
                    const d = new Date(s.releaseDate);
                    if (!isNaN(d) && (now - d) <= THIRTY_DAYS) newShows.push(s);
                    else oldShows.push(s);
                });
            }
        %>

        <section id="latest-updates">
            <h2>Latest Updates</h2>
            <% if (newShows && newShows.length) { %>
                <ul>
                    <% newShows.forEach(show => { %>
                        <li>
                            <a href="/shows/<%= show.id %>"><%= show.title %></a> - <%= show.genre %>
                            <% if (show.releaseDate) { %> (<%= new Date(show.releaseDate).toLocaleDateString() %>)<% } %>
                        </li>
                    <% }); %>
                </ul>
            <% } else { %>
                <p>No recent updates. Check back soon.</p>
            <% } %>
        </section>

        <section id="older-dramas">
            <h2>Older Dramas — Page 1</h2>
            <% if (oldShows && oldShows.length) { %>
                <ul>
                    <% oldShows.slice(0,10).forEach(show => { %>
                        <li>
                            <a href="/shows/<%= show.id %>"><%= show.title %></a> - <%= show.genre %>
                        </li>
                    <% }); %>
                </ul>
                <% if (oldShows.length > 10) { %>
                    <a href="/shows?page=2">View more (page 2)</a>
                <% } %>
            <% } else { %>
                <p>No older dramas available.</p>
            <% } %>
        </section>
